package infrapunaohjaus;

import lejos.hardware.Button;
import lejos.hardware.lcd.LCD;
import lejos.hardware.port.SensorPort;
import lejos.hardware.sensor.EV3TouchSensor;
import lejos.utility.Delay;

import java.util.Date;

import javax.swing.plaf.basic.BasicButtonListener;

public class Kosketus {
/**
 * metodi, jolla voidaan lopettaa ohjelman ajo käyttäen
 * kosketusanturia.
 */
	public void kosketetaan() {
/**
 * metodimuuttujana on painallukset
 */
		int painallukset = 0;
		/**
		 * äynnistetään kosketusanturi 
		 */
		EV3TouchSensor ts = new EV3TouchSensor(SensorPort.S2);
		float[] sample = new float[ts.sampleSize()];
		while (true) {
			/**
			 * luetaan kosketusanturia loopissa ja mikäli
			 * sensoria painetaan, ohjelma lisää muuttujaan ykkösen, 
			 * tämän jälkeen looppi hajoaa ja ohjelma siirtyy seuraavaan 
			 * vaiheeseen jossa System.exit(); komento sulkee koko ohjelman.
			 */
			ts.getTouchMode();
			ts.fetchSample(sample, 0);
			if (sample[0] != 0) {
				painallukset = painallukset + 1;
				break;
			}
		}

		LCD.drawString("Suljetaan...", 0, 3);
		Delay.msDelay(1000);
		System.exit(1);

	}
}
